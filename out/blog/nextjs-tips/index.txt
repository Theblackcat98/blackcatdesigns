const HeavyComponent = dynamic(() => import('@/components/Heavy'), {

})

export default function Page() {
  return &#x3C;HeavyComponent />
}
</code></pre>

export default function Hero() {

      src="/hero.jpg"
      alt="Hero image"
      width={1200}
      height={600}
      priority
    />
  )
}
</code></pre>
<p>Benefits:</p>
<ul>
<li>Automatic format selection (WebP, AVIF)</li>
<li>Responsive image sizing</li>
<li>Lazy loading by default</li>
<li>Prevention of Cumulative Layout Shift (CLS)</li>
</ul>

import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

  if (!token &#x26;&#x26; request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }
}

export const config = {
  matcher: ['/dashboard/:path*'],
}

function Skeleton() {

}

  const posts = await fetchPosts()

}

export default function Page() {
  return (

      &#x3C;Posts />
    &#x3C;/Suspense>
  )
}
</code></pre>

</ul>
